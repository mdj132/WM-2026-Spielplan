// ─── HELPERS ─────────────────────────────────────────────────────────────────
const dt = (date, time) => new Date(`${date}T${time.length===5?time+":00":time}`);
const canTip = (date, time) => new Date() < new Date(dt(date,time).getTime() - 5*60*1000);
const isStarted = (date, time) => new Date() >= dt(date, time);
const isFinished = (date, time) => new Date() >= new Date(dt(date,time).getTime() + 110*60*1000);
const fmtDate = d => new Date(d).toLocaleDateString("de-DE",{weekday:"short",day:"2-digit",month:"2-digit"});

function calcPoints(tip, result) {
  if (!tip || !result) return 0;
  const tH=tip.home, tA=tip.away, rH=result.home, rA=result.away;
  if (tH===rH && tA===rA) return 5;
  const tDiff=tH-tA, rDiff=rH-rA;
  const sameTend = (tDiff>0&&rDiff>0)||(tDiff<0&&rDiff<0)||(tDiff===0&&rDiff===0);
  if (!sameTend) return 0;
  if (tDiff===rDiff) return 3; // same goal difference, right tendency
  return 2; // right tendency only
}

function calcGroupTable(group, results) {
  const teams = GROUP_TEAMS[group];
  const stats = Object.fromEntries(teams.map(t=>[t,{sp:0,g:0,u:0,v:0,tore:0,gegentore:0,punkte:0}]));
  GAMES_DATA.filter(g=>g.group===group).forEach(g=>{
    const r = results[g.id];
    if (!r) return;
    const h=r.home, a=r.away;
    stats[g.home].sp++; stats[g.away].sp++;
    stats[g.home].tore+=h; stats[g.home].gegentore+=a;
    stats[g.away].tore+=a; stats[g.away].gegentore+=h;
    if (h>a) { stats[g.home].g++; stats[g.home].punkte+=3; stats[g.away].v++; }
    else if (h<a) { stats[g.away].g++; stats[g.away].punkte+=3; stats[g.home].v++; }
    else { stats[g.home].u++; stats[g.away].u++; stats[g.home].punkte++; stats[g.away].punkte++; }
  });
  return Object.entries(stats)
    .map(([name,s])=>({name,...s,td:s.tore-s.gegentore}))
    .sort((a,b)=>b.punkte-a.punkte||b.td-a.td||b.tore-a.tore);
}

                <td style={{...td,color:t.td>0?"#16a34a":t.td<0?"#dc2626":"#64748b"}}>{t.td>0?"+":""}{t.td}</td>
                <td style={{...td,fontWeight:700,color:"#2563eb"}}>{t.punkte}</td>
              </tr>
            ))}
          </tbody>
        </table>
      </div>
    </div>
  );
}

function KicktippTable({ users, allTips, results }) {
  const scored = users.map(u => {
    let pts = 0, exact=0, tendency=0;
    GAMES_DATA.forEach(g => {
      const t = allTips[u.id]?.[g.id];
      const r = results[g.id];
      const p = calcPoints(t, r);
      pts += p;
      if (p===5) exact++;
      else if (p>=2) tendency++;
    });
    return { ...u, pts, exact, tendency };
  }).sort((a,b)=>b.pts-a.pts||b.exact-a.exact);

  if (scored.length===0) return (
    <div style={{textAlign:"center",color:"#94a3b8",padding:"32px 0",fontSize:"0.85rem"}}>
      Noch keine Teilnehmer registriert.
    </div>
  );

  const th={fontSize:"0.65rem",color:"#64748b",fontWeight:600,padding:"6px 8px",textAlign:"center"};
  const td={fontSize:"0.8rem",padding:"7px 8px",textAlign:"center",color:"#1e3a5f"};

  return (
    <div style={{background:"#fff",border:"1px solid #e2e8f0",borderRadius:10,overflow:"hidden",boxShadow:"0 1px 4px rgba(30,58,95,0.05)"}}>
      <table style={{width:"100%",borderCollapse:"collapse"}}>
        <thead>
          <tr style={{background:"#1e3a5f"}}>
            <th style={{...th,color:"#93c5fd",textAlign:"left",paddingLeft:12}}>#</th>
            <th style={{...th,color:"#93c5fd",textAlign:"left"}}>Name</th>
            <th style={{...th,color:"#93c5fd"}}>Exakt</th>
            <th style={{...th,color:"#93c5fd"}}>Tendenz</th>
            <th style={{...th,color:"#fff",fontWeight:700}}>Punkte</th>
          </tr>
        </thead>
        <tbody>
          {scored.map((u,i)=>(
            <tr key={u.id} style={{borderTop:"1px solid #f1f5f9",background:i===0?"#fefce8":i===1?"#f8fafc":i===2?"#fff7ed":"#fff"}}>
              <td style={{...td,paddingLeft:12,fontWeight:700,color:i===0?"#d97706":i===1?"#64748b":i===2?"#b45309":"#94a3b8"}}>
                {i===0?"🥇":i===1?"🥈":i===2?"🥉":i+1}
              </td>
              <td style={{...td,textAlign:"left",fontWeight:600}}>{u.name}</td>
              <td style={{...td,color:"#16a34a",fontWeight:600}}>{u.exact}x</td>
              <td style={{...td,color:"#2563eb"}}>{u.tendency}x</td>
              <td style={{...td,fontWeight:800,fontSize:"0.95rem",color:"#2563eb"}}>{u.pts}</td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}

function RegisterForm({ onRegister }) {
  const [name,setName]=useState("");
  const [email,setEmail]=useState("");
  const [err,setErr]=useState("");
  const inp={width:"100%",padding:"10px 12px",border:"1px solid #cbd5e1",borderRadius:8,fontSize:"0.9rem",color:"#1e3a5f",outline:"none",boxSizing:"border-box"};
  const submit = () => {
    if (!name.trim()) return setErr("Bitte Namen eingeben.");
    if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) return setErr("Bitte gültige E-Mail eingeben.");
    onRegister(name.trim(), email.trim());
  };
  return (
    <div style={{background:"#fff",border:"1px solid #bfdbfe",borderRadius:14,padding:"24px",maxWidth:380,margin:"0 auto",boxShadow:"0 4px 20px rgba(30,58,95,0.1)"}}>
      <h3 style={{margin:"0 0 6px",color:"#1e3a5f",fontSize:"1.1rem"}}>🏆 Kicktipp beitreten</h3>
      <p style={{margin:"0 0 18px",fontSize:"0.8rem",color:"#64748b"}}>Name und E-Mail eintragen und automatisch mitspielen.</p>
      <div style={{marginBottom:12}}>
        <label style={{fontSize:"0.78rem",color:"#64748b",display:"block",marginBottom:4}}>Name</label>
        <input value={name} onChange={e=>{setName(e.target.value);setErr("");}} style={inp} placeholder="z.B. Max Mustermann"/>
      </div>
      <div style={{marginBottom:16}}>
        <label style={{fontSize:"0.78rem",color:"#64748b",display:"block",marginBottom:4}}>E-Mail</label>
        <input value={email} onChange={e=>{setEmail(e.target.value);setErr("");}} style={inp} placeholder="max@beispiel.de" type="email"/>
      </div>
      {err && <div style={{color:"#ef4444",fontSize:"0.78rem",marginBottom:10}}>{err}</div>}
      <button onClick={submit} style={{background:"#2563eb",border:"none",color:"#fff",padding:"10px 20px",borderRadius:8,cursor:"pointer",fontWeight:700,fontSize:"0.9rem",width:"100%"}}>
        Registrieren & mitspielen
      </button>
      <p style={{margin:"10px 0 0",fontSize:"0.7rem",color:"#94a3b8",textAlign:"center"}}>Deine Daten werden nur lokal gespeichert.</p>
    </div>
  );
}

// ─── MAIN APP ─────────────────────────────────────────────────────────────────
export default function App() {
  const [tab, setTab] = useState("Spielplan");
  const [activeGroup, setActiveGroup] = useState("Alle");
  const [results, setResults] = useState({});
  const [allTips, setAllTips] = useState({}); // { userId: { gameId: {home,away} } }
  const [users, setUsers] = useState([]); // [{id,name,email}]
  const [currentUser, setCurrentUser] = useState(null);
  const [modalGame, setModalGame] = useState(null);
  const [loading, setLoading] = useState(false);
  const [lastFetch, setLastFetch] = useState(null);

  // Load from storage
  useEffect(()=>{
    (async()=>{
      try { const r=await window.storage.get("wm26_results"); if(r) setResults(JSON.parse(r.value)); } catch{}
      try { const t=await window.storage.get("wm26_tips"); if(t) setAllTips(JSON.parse(t.value)); } catch{}
      try { const u=await window.storage.get("wm26_users"); if(u) setUsers(JSON.parse(u.value)); } catch{}
      try { const cu=await window.storage.get("wm26_currentUser"); if(cu) setCurrentUser(JSON.parse(cu.value)); } catch{}
    })();
  },[]);

  // Auto-refresh results every 5 min if games are running
  useEffect(()=>{
    const iv = setInterval(()=>{
      const running = GAMES_DATA.some(g=>isStarted(g.date,g.time)&&!isFinished(g.date,g.time));
      if (running) fetchResults(true);
    }, 5*60*1000);
    return ()=>clearInterval(iv);
  },[]);

  const saveResults = async (r) => {
    setResults(r);
    try { await window.storage.set("wm26_results", JSON.stringify(r)); } catch{}
  };

  const fetchResults = async (silent=false) => {
    if (!silent) setLoading(true);
    try {
      const started = GAMES_DATA.filter(g=>isStarted(g.date,g.time));
      if (!started.length) { setLoading(false); return; }
      const res = await fetch("https://api.anthropic.com/v1/messages",{
        method:"POST",headers:{"Content-Type":"application/json"},
        body:JSON.stringify({
          model:"claude-sonnet-4-20250514",max_tokens:1500,
          system:`Du bist ein Fußball-Assistent. Antworte NUR mit einem JSON-Objekt. Kein Markdown, keine Erklärungen. Format: {"results":[{"id":<number>,"home":<number>,"away":<number>}]}. Nur bereits abgeschlossene Spiele zurückgeben. Heute: ${new Date().toLocaleDateString("de-DE")}.`,
          tools:[{type:"web_search_20250305",name:"web_search"}],
          messages:[{role:"user",content:`Suche die aktuellen Ergebnisse der FIFA WM 2026. Gib mir alle abgeschlossenen Spiele als JSON:\n${started.map(g=>`ID ${g.id}: ${g.home} vs ${g.away} (${g.date} ${g.time})`).join("\n")}\nNur JSON zurückgeben.`}]
        })
      });
      const data = await res.json();
      const text = data.content.filter(b=>b.type==="text").map(b=>b.text).join("");
      const parsed = JSON.parse(text.replace(/```json|```/g,"").trim());
      if (parsed.results) {
        const nr={...results};
        parsed.results.forEach(r=>{nr[r.id]={home:r.home,away:r.away};});
        await saveResults(nr);
        setLastFetch(new Date());
      }
    } catch(e){console.error(e);}
    setLoading(false);
  };

  const register = async (name, email) => {
    const id = `${name.replace(/\s+/g,"_")}_${Date.now()}`;
    const user = {id,name,email};
    const newUsers = [...users.filter(u=>u.email!==email), user];
    setUsers(newUsers);
    setCurrentUser(user);
    try { await window.storage.set("wm26_users", JSON.stringify(newUsers)); } catch{}
    try { await window.storage.set("wm26_currentUser", JSON.stringify(user)); } catch{}
  };

  const saveTip = async (game, tip) => {
    const uid = currentUser.id;
    const newTips = {...allTips, [uid]:{...(allTips[uid]||{}), [game.id]:tip}};
    setAllTips(newTips);
    try { await window.storage.set("wm26_tips", JSON.stringify(newTips)); } catch{}
    setModalGame(null);
  };

  const logout = async () => {
    setCurrentUser(null);
    try { await window.storage.delete("wm26_currentUser"); } catch{}
  };

  const myTips = currentUser ? (allTips[currentUser.id]||{}) : {};
  const displayGames = activeGroup==="Alle" ? GAMES_DATA : GAMES_DATA.filter(g=>g.group===activeGroup);

  // Build per-game allTips map: { gameId: { userName: tip } }
  const gameTipsByUser = {};
  users.forEach(u=>{
    const ut = allTips[u.id]||{};
    Object.entries(ut).forEach(([gid,tip])=>{
      if (!gameTipsByUser[gid]) gameTipsByUser[gid]={};
      gameTipsByUser[gid][u.name]=tip;
    });
  });

  const tabStyle = (t) => ({
    background:"none",border:"none",
    borderBottom:tab===t?"3px solid #2563eb":"3px solid transparent",
    color:tab===t?"#2563eb":"#64748b",
    padding:"11px 0",flex:1,cursor:"pointer",
    fontWeight:tab===t?700:400,fontSize:"0.82rem",
    transition:"all 0.15s"
  });

  return (
    <div style={{minHeight:"100vh",background:"#f1f5f9",fontFamily:"system-ui,-apple-system,sans-serif",color:"#1e3a5f"}}>
      {/* Header */}
      <div style={{background:"linear-gradient(135deg,#1e3a5f,#2563eb)",padding:"16px 16px 12px",boxShadow:"0 2px 12px rgba(30,58,95,0.2)"}}>
        <div style={{maxWidth:680,margin:"0 auto",display:"flex",justifyContent:"space-between",alignItems:"flex-start",gap:10}}>
          <div>
            <h1 style={{margin:0,fontSize:"1.35rem",fontWeight:800,color:"#fff"}}>⚽ WM 2026</h1>
            <p style={{margin:"2px 0 0",fontSize:"0.72rem",color:"#bfdbfe"}}>11. Juni – 19. Juli · Alle Zeiten MESZ</p>
          </div>
          <div style={{textAlign:"right",flexShrink:0}}>
            {currentUser ? (
              <div>
                <div style={{fontSize:"0.72rem",color:"#fff",background:"rgba(255,255,255,0.15)",borderRadius:8,padding:"3px 8px",marginBottom:5}}>👤 {currentUser.name}</div>
                <div style={{display:"flex",gap:5,justifyContent:"flex-end"}}>
                  <button onClick={()=>fetchResults()} disabled={loading} style={{background:"rgba(255,255,255,0.2)",border:"1px solid rgba(255,255,255,0.3)",color:"#fff",padding:"4px 10px",borderRadius:6,cursor:loading?"not-allowed":"pointer",fontSize:"0.7rem",fontWeight:700}}>
                    {loading?"…":"🔄"}
                  </button>
                  <button onClick={logout} style={{background:"rgba(255,255,255,0.1)",border:"1px solid rgba(255,255,255,0.2)",color:"#bfdbfe",padding:"4px 10px",borderRadius:6,cursor:"pointer",fontSize:"0.7rem"}}>Abmelden</button>
                </div>
                {lastFetch && <div style={{fontSize:"0.62rem",color:"#bfdbfe",marginTop:3}}>Stand: {lastFetch.toLocaleTimeString("de-DE",{hour:"2-digit",minute:"2-digit"})}</div>}
              </div>
            ) : (
              <button onClick={()=>fetchResults()} disabled={loading} style={{background:"rgba(255,255,255,0.2)",border:"1px solid rgba(255,255,255,0.3)",color:"#fff",padding:"5px 12px",borderRadius:7,cursor:loading?"not-allowed":"pointer",fontSize:"0.72rem",fontWeight:700}}>
                {loading?"Lade…":"🔄 Aktualisieren"}
              </button>
            )}
          </div>
        </div>
      </div>

      {/* Tabs */}
      <div style={{background:"#fff",borderBottom:"1px solid #e2e8f0",position:"sticky",top:0,zIndex:100}}>
        <div style={{maxWidth:680,margin:"0 auto",display:"flex"}}>
          {TABS.map(t=><button key={t} style={tabStyle(t)} onClick={()=>setTab(t)}>{t}</button>)}
        </div>
      </div>

      <div style={{maxWidth:680,margin:"0 auto",padding:"12px 12px 32px"}}>

        {/* ── SPIELPLAN ── */}
        {tab==="Spielplan" && (
          <>
            {/* Group filter */}
            <div style={{overflowX:"auto",marginBottom:10,paddingBottom:2}}>
              <div style={{display:"flex",gap:4,minWidth:"max-content"}}>
                {["Alle",...GROUPS].map(g=>(
                  <button key={g} onClick={()=>setActiveGroup(g)} style={{background:activeGroup===g?"#2563eb":"#fff",border:"1px solid",borderColor:activeGroup===g?"#2563eb":"#e2e8f0",color:activeGroup===g?"#fff":"#64748b",padding:"4px 10px",borderRadius:99,cursor:"pointer",fontSize:"0.72rem",fontWeight:activeGroup===g?700:400,whiteSpace:"nowrap"}}>

                <GameCard key={game.id} game={game} myTip={myTips[game.id]}

              ))}
            </div>
          </>
        )}

        {/* ── GRUPPEN ── */}
        {tab==="Gruppen" && (
          <div style={{display:"flex",flexDirection:"column",gap:14}}>
            {GROUPS.map(g=><GroupTable key={g} group={g} results={results}/>)}
          </div>
        )}

        {/* ── KICKTIPP ── */}
        {tab==="Kicktipp" && (
          <div style={{display:"flex",flexDirection:"column",gap:14}}>
            {!currentUser && (
              <RegisterForm onRegister={register}/>

                ✅ Du nimmst als <strong>{currentUser.name}</strong> teil. Gib im Spielplan deine Tipps ab!
              </div>
            )}

              <h3 style={{margin:"0 0 10px",fontSize:"0.95rem",color:"#1e3a5f",fontWeight:700}}>🏆 Gesamtwertung</h3>

        <TipModal game={modalGame} onClose={()=>setModalGame(null)} onSave={(tip)=>saveTip(modalGame,tip)} existingTip={myTips[modalGame.id]}/>
      )}

  );
}

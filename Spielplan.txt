function parseDateTime(date, time) {
  return new Date(`${date}T${time}:00`);
}
function canTip(date, time) {
  const kickoff = parseDateTime(date, time);
  return new Date() < new Date(kickoff.getTime() - 5 * 60 * 1000);
}
function isStarted(date, time) {
  return new Date() >= parseDateTime(date, time);
}
function formatDate(date) {
  return new Date(date).toLocaleDateString("de-DE", { weekday:"short", day:"2-digit", month:"2-digit" });
}

        })
      });
      const data = await res.json();
      const text = data.content.filter(b=>b.type==="text").map(b=>b.text).join("");
      const parsed = JSON.parse(text.replace(/```json|```/g,"").trim());
      if (parsed.results) {
        const newR = {...results};
        parsed.results.forEach(r=>{newR[r.id]={home:r.home,away:r.away};});
        setResults(newR);
        try { await window.storage.set("wm2026_results", JSON.stringify(newR)); } catch {}
      }
    } catch(e) { console.error(e); }
    setLoading(false);
  };

  const groups = ["Alle","A","B","C","D","E","F","G","H","I","J","K","L"];
  let displayGames = [];
  if (activePhase === "Gruppenphase") displayGames = activeGroup === "Alle" ? GAMES_DATA : GAMES_DATA.filter(g=>g.group===activeGroup);
  else if (activePhase === "Sechzehntelfinale") displayGames = ROUND_OF_32;
  else displayGames = KNOCKOUT.filter(g=>g.round===activePhase);

  const totalTips = Object.keys(tips).length;

          </div>
          <div style={{ textAlign:"right" }}>
            {totalTips > 0 && (
              <div style={{ fontSize:"0.75rem", color:"#fff", background:"rgba(255,255,255,0.15)", borderRadius:8, padding:"4px 10px", marginBottom:6 }}>

              </div>

            <button onClick={fetchResults} disabled={loading} style={{ background:loading?"rgba(255,255,255,0.15)":"rgba(255,255,255,0.2)", border:"1px solid rgba(255,255,255,0.3)", color:"#fff", padding:"5px 12px", borderRadius:7, cursor:loading?"not-allowed":"pointer", fontSize:"0.75rem", fontWeight:700 }}>

        <div style={{ display:"flex", maxWidth:700, margin:"0 auto" }}>
          {PHASES.map(p => (
            <button key={p} onClick={()=>setActivePhase(p)} style={{ background:"none", border:"none", borderBottom:activePhase===p?"2px solid #2563eb":"2px solid transparent", color:activePhase===p?"#2563eb":"#64748b", padding:"10px 14px", cursor:"pointer", fontWeight:activePhase===p?700:400, fontSize:"0.78rem", whiteSpace:"nowrap" }}>

      {/* Group filter */}
      {activePhase === "Gruppenphase" && (

            {groups.map(g => (
              <button key={g} onClick={()=>setActiveGroup(g)} style={{ background:activeGroup===g?"#2563eb":"#fff", border:"1px solid", borderColor:activeGroup===g?"#2563eb":"#e2e8f0", color:activeGroup===g?"#fff":"#64748b", padding:"3px 10px", borderRadius:99, cursor:"pointer", fontSize:"0.75rem", fontWeight:activeGroup===g?700:400, whiteSpace:"nowrap" }}>
                {g === "Alle" ? "Alle" : `Gr. ${g}`}
              </button>
            ))}
          </div>
        </div>
      )}

      {/* Games */}
      <div style={{ maxWidth:700, margin:"0 auto", padding:"16px 12px", display:"flex", flexDirection:"column", gap:8 }}>
        {displayGames.length === 0 && (
          <div style={{ textAlign:"center", color:"#94a3b8", padding:"40px 0", fontSize:"0.9rem" }}>Noch keine Spiele in dieser Runde bekannt.</div>
        )}
        {displayGames.map(game => (
          <GameCard key={game.id} game={game} tip={tips[game.id]} result={results[game.id]} onTip={setModalGame} />
        ))}
      </div>

      {modalGame && (
        <TipModal game={modalGame} onClose={()=>setModalGame(null)} onSave={(tip)=>saveTip(modalGame,tip)} existingTip={tips[modalGame.id]} />
      )}
    </div>
  );
}
